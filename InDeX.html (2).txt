                </tr>
                <tr>
                    <td><strong>Coffee</strong></td>
                    <td>Coffee Wilt Disease</td>
                    <td>Wilting, leaf drop</td>

                </tr>
                <tr>
                    <td><strong>Cassava</strong></td>
